#### 节流函数的基本实现

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body>
    <input type="text" />
    <script>
      function mydebounce(fn, delay) {
        // 记录上一次的timer
        let timer = null;
        // 返回需要执行的函数
        return function _debounce() {
          //频繁触发执行都会先清除上一次的timer
          if (timer) clearTimeout(timer);

          timer = setTimeout(() => {
            fn();
            timer = null; // 执行过函数之后, 将timer重新置null
          }, delay);
        };
      }
    </script>
    <script>
      const inputEl = document.querySelector("input");
      inputEl.oninput = mydebounce(function (e) {
        console.log(e, "----------");
      }, 1000);
    </script>
  </body>
</html>

```

#### 获取返回值的实现

获取返回值的方式有两种

传入回调函数

通过promise.then来获取

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body>
    <input type="text" />
    <button>取消</button>
    <script>
      function mydebounce(
        fn,
        delay,
        immediate = false,
        resultCallback = () => {}
      ) {
        // 记录上一次的timer
        let timer = null;
        // 记录是否需要第一次是否执行和immediate配合
        let isInvoke = false;
        // 返回需要执行的函数
        function _debounce(...args) {
          return new Promise((resolve, reject) => {
            try {
              //频繁触发执行都会先清除上一次的timer
              if (timer) clearTimeout(timer);
              // 当immediate为true时
              if (immediate && !isInvoke) {
                const result = fn.apply(this, args);
                resultCallback(result);
                resolve(result);
                isInvoke = true;
                return;
              }

              timer = setTimeout(() => {
                const result = fn.apply(this, args);
                resultCallback(result);
                resolve(result);
                timer = null; // 执行过函数之后, 将timer重新置null
                isInvoke = false; //重置
              }, delay);
            } catch (error) {
              reject(error);
            }
          });
        }
        // 取消功能
        _debounce.cancel = function () {
          if (timer) clearTimeout(timer);
          console.log("button");
          timer = null;
          isInvoke = false;
        };
        return _debounce;
      }
    </script>
    <script>
      const inputEl = document.querySelector("input");
      const buttonEl = document.querySelector("button");
      const debounce = mydebounce(
        function (e) {
          console.log(e, this, "----------");
        },
        3000,
        true
      );

      inputEl.oninput = debounce;

      buttonEl.onclick = debounce.cancel;
    </script>
  </body>
</html>

```
